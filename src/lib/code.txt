DemoController.java
package com.user.entitlement.controller;

import com.user.entitlement.model.*;
import com.user.entitlement.repository.*;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

import java.util.List;

@RestController
@RequestMapping("/api")
public class DemoController {

    @Autowired private ApplicationMstRepository applicationRepo;
    @Autowired private ApplicationProfileDetailsRepository appProfileRepo;
    @Autowired private RoleMstRepository roleRepo;
    @Autowired private RoleProfileDetailsRepository roleProfileRepo;
    @Autowired private BusinessUsersMstRepository businessUserRepo;
    @Autowired private BusinessUserProfileDetailsRepository businessUserProfileRepo;

    @GetMapping("/applications")
    public List<ApplicationMst> getAllApplications() {
        return applicationRepo.findAll();
    }

    @GetMapping("/application-profiles")
    public List<ApplicationProfileDetails> getAllApplicationProfiles() {
        return appProfileRepo.findAll();
    }

    @GetMapping("/roles")
    public List<RoleMst> getAllRoles() {
        return roleRepo.findAll();
    }

    @GetMapping("/role-profiles")
    public List<RoleProfileDetails> getAllRoleProfiles() {
        return roleProfileRepo.findAll();
    }

    @GetMapping("/business-users")
    public List<BusinessUsersMst> getAllBusinessUsers() {
        return businessUserRepo.findAll();
    }

    @GetMapping("/business-user-profiles")
    public List<BusinessUserProfileDetails> getAllBusinessUserProfiles() {
        return businessUserProfileRepo.findAll();
    }
}
